Create Database ParcelDB

Use ParcelDB


CREATE TABLE PaymentHistory1 (
    ID INT IDENTITY(1,1) PRIMARY KEY, 
    ParcelID VARCHAR(20),            
    LastPaymentAmount VARCHAR(20),
    LastPaymentDate VARCHAR(20),        
    FiscalTaxYearPayments VARCHAR(20),
    PriorCalendarYearPayments VARCHAR(20), 
    CurrentCalendarYearPayments VARCHAR(20), 
);

SELECT TOP (1000) [ID]
      ,[ParcelID]
      ,[LastPaymentAmount]
      ,[LastPaymentDate]
      ,[FiscalTaxYearPayments]
      ,[PriorCalendarYearPayments]
      ,[CurrentCalendarYearPayments]
  FROM [ParcelDB].[dbo].[PaymentHistory1]
